#include <iostream>
#include <vector>
#include <queue>

using namespace std;

class Solution {
public:
    vector<vector<int>> kClosest(vector<vector<int>>& points, int k) {
        // Max-Heap: almacena pares de {distancia_cuadrada, índice_del_punto}
        priority_queue<pair<int, int>> maxHeap;

        for (int i = 0; i < points.size(); ++i) {
            int dist = points[i][0] * points[i][0] + points[i][1] * points[i][1];
            maxHeap.push({dist, i});

            // Si sobrepasamos k, eliminamos el punto con mayor distancia
            if (maxHeap.size() > k) {
                maxHeap.pop();
            }
        }

        vector<vector<int>> result;
        while (!maxHeap.empty()) {
            result.push_back(points[maxHeap.top().second]);
            maxHeap.pop();
        }
        return result;
    }
};

int main() {
    Solution sol;
    
    // Datos del ejemplo en la imagen
    vector<vector<int>> points = {{1, 3}, {-2, 2}};
    int k = 1;

    vector<vector<int>> result = sol.kClosest(points, k);

    // Imprimir los resultados
    cout << "Los " << k << " puntos mas cercanos al origen son:" << endl;
    for (const auto& point : result) {
        cout << "[" << point[0] << ", " << point[1] << "]" << endl;
    }

    return 0;
}
-----------------------Rectangle Overlap-----------------------
#include <iostream>
#include <vector>

using namespace std;

class Solution {
public:
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        // Un rectángulo se define como [x1, y1, x2, y2]
        // NO se superponen si uno está totalmente a la izquierda, derecha, arriba o abajo del otro.
        
        // Usamos la negación de todas las condiciones de NO superposición:
        return !(rec1[2] <= rec2[0] || // rec1 está a la izquierda de rec2
                 rec1[0] >= rec2[2] || // rec1 está a la derecha de rec2
                 rec1[1] >= rec2[3] || // rec1 está arriba de rec2
                 rec1[3] <= rec2[1]);  // rec1 está abajo de rec2
    }
};

int main() {
    Solution sol;

    // Ejemplo 1: Superposición positiva
    vector<int> rec1_ex1 = {0, 0, 2, 2};
    vector<int> rec2_ex1 = {1, 1, 3, 3};
    cout << "Ejemplo 1: " << (sol.isRectangleOverlap(rec1_ex1, rec2_ex1) ? "true" : "false") 
         << " (Esperado: true)" << endl;

    // Ejemplo 2: Solo se tocan en el borde (x=1)
    vector<int> rec1_ex2 = {0, 0, 1, 1};
    vector<int> rec2_ex2 = {1, 0, 2, 1};
    cout << "Ejemplo 2: " << (sol.isRectangleOverlap(rec1_ex2, rec2_ex2) ? "true" : "false") 
         << " (Esperado: false)" << endl;

    // Ejemplo 3: Totalmente separados
    vector<int> rec1_ex3 = {0, 0, 1, 1};
    vector<int> rec2_ex3 = {2, 2, 3, 3};
    cout << "Ejemplo 3: " << (sol.isRectangleOverlap(rec1_ex3, rec2_ex3) ? "true" : "false") 
         << " (Esperado: false)" << endl;

    return 0;
}
-----------------------------Minimum Area Rectangle----------------------------------
#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    int minAreaRect(vector<vector<int>>& points) {
        // Usamos un set de pares para búsqueda rápida O(log n)
        // O podrías usar un hash de string "x,y" para O(1)
        set<pair<int, int>> pointSet;
        for (auto& p : points) {
            pointSet.insert({p[0], p[1]});
        }

        int minArea = INT_MAX;
        int n = points.size();

        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                int x1 = points[i][0], y1 = points[i][1];
                int x2 = points[j][0], y2 = points[j][1];

                // Para ser diagonal de un rectángulo paralelo a los ejes:
                // Los puntos no deben compartir ni X ni Y.
                if (x1 != x2 && y1 != y2) {
                    // Verificamos si existen los otros dos vértices
                    if (pointSet.count({x1, y2}) && pointSet.count({x2, y1})) {
                        int area = abs(x1 - x2) * abs(y1 - y2);
                        minArea = min(minArea, area);
                    }
                }
            }
        }

        return (minArea == INT_MAX) ? 0 : minArea;
    }
};

int main() {
    Solution sol;

    // Ejemplo 1 de la imagen
    // Puntos: [1,1], [1,3], [3,1], [3,3], [2,2]
    vector<vector<int>> p1 = {{1, 1}, {1, 3}, {3, 1}, {3, 3}, {2, 2}};
    cout << "Entrada: [[1,1],[1,3],[3,1],[3,3],[2,2]]" << endl;
    cout << "Salida: " << sol.minAreaRect(p1) << " (Esperado: 4)" << endl;

    return 0;
}
--------------------------queries-on-number-of-points-------------------------------
#include <iostream>
#include <vector>

using namespace std;

class Solution {
public:
    vector<int> countPoints(vector<vector<int>>& points, vector<vector<int>>& queries) {
        vector<int> answer;
        
        // Iteramos por cada consulta (cada círculo)
        for (const auto& query : queries) {
            int qx = query[0];
            int qy = query[1];
            int qr = query[2];
            int rSquared = qr * qr; // Radio al cuadrado
            
            int count = 0;
            // Para cada círculo, revisamos todos los puntos
            for (const auto& point : points) {
                int px = point[0];
                int py = point[1];
                
                // Calculamos distancia al cuadrado: (x2-x1)^2 + (y2-y1)^2
                int distSquared = (px - qx) * (px - qx) + (py - qy) * (py - qy);
                
                if (distSquared <= rSquared) {
                    count++;
                }
            }
            answer.push_back(count);
        }
        
        return answer;
    }
};

int main() {
    Solution sol;

    // Datos del Ejemplo 1 de la imagen
    vector<vector<int>> points = {{1, 3}, {3, 3}, {5, 3}, {2, 2}};
    vector<vector<int>> queries = {{2, 3, 1}, {4, 3, 1}, {1, 1, 2}};

    vector<int> result = sol.countPoints(points, queries);

    // Mostrar resultados
    cout << "Resultado: [";
    for (int i = 0; i < result.size(); ++i) {
        cout << result[i] << (i == result.size() - 1 ? "" : ", ");
    }
    cout << "]" << endl;
    cout << "Esperado:  [3, 2, 2]" << endl;

    return 0;
}
